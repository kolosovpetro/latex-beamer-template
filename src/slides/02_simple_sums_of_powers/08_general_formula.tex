\begin{frame}
    In general,
    \begin{align*}
        \sum_{k=1}^{n} k^m &= \sum_{j=0}^{m} T(t, m,j) \left[ \binom{n-t+1}{j+1} + (-1)^j \binom{j+t-1}{j+1} \right]
    \end{align*}
    where
    \begin{align*}
        T(t,m,j) = \sum_{k=0}^{m} \binom{t}{k} \stirlingii{m}{j+k} (j+k)!
    \end{align*}
\end{frame}
