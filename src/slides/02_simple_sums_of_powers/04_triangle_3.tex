        \KnuthRFoldSum{1}{n}{2} &= 9  \left( \binom{n-2}{1} + \binom{2}{1}  \right) + 7  \left( \binom{n-2}{2} - \binom{3}{2}  \right) + 2 \left( \binom{n-2}{3} + \binom{4}{3}  \right) \\
        \KnuthRFoldSum{1}{n}{3} &= 27 \left( \binom{n-2}{1} + \binom{2}{1}  \right) + 37 \left( \binom{n-2}{2} - \binom{3}{2}  \right) \\
        &+ 24 \left( \binom{n-2}{3} + \binom{4}{3}  \right) + 6  \left( \binom{n-2}{4} - \binom{5}{4}  \right) \\
        \KnuthRFoldSum{1}{n}{m} &= \sum_{j=0}^{m} T(3,m,j) \left[ \binom{n-2}{j+1} + (-1)^j \binom{j+2}{j+1}  \right]
