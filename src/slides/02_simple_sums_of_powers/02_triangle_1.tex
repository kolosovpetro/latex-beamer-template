    \begin{align*}
        \KnuthRFoldSum{1}{n}{0} &= 1 \left( \binom{n}{1} + \binom{0}{1} \right) \\
        \KnuthRFoldSum{1}{n}{1} &= 1 \left( \binom{n}{1} + \binom{0}{1} \right) + 1 \left( \binom{n}{2} - \binom{1}{2} \right) \\
        \KnuthRFoldSum{1}{n}{2} &= 1 \left( \binom{n}{1} + \binom{0}{1} \right) + 3 \left( \binom{n}{2} - \binom{1}{2} \right)  + 2  \left( \binom{n}{3} + \binom{2}{3} \right)  \\
        \KnuthRFoldSum{1}{n}{3} &= 1 \left( \binom{n}{1} + \binom{0}{1} \right) + 7 \left( \binom{n}{2} - \binom{1}{2} \right)  + 12 \left( \binom{n}{3} + \binom{2}{3} \right) + 6 \left( \binom{n}{4} - \binom{3}{4} \right) \\
        \KnuthRFoldSum{1}{n}{m} &= \sum_{j=0}^{m} T(1,m,j) \left[ \binom{n}{j+1} + (-1)^j \binom{j}{j+1} \right]
    \end{align*}
