        \KnuthRFoldSum{1}{n}{1} &= 0 \left( \binom{n+1}{1} + \binom{-1}{1} \right) + 1  \left( \binom{n+1}{2} - \binom{0}{2} \right) \\
        \KnuthRFoldSum{1}{n}{2} &= 0 \left( \binom{n+1}{1} + \binom{-1}{1} \right) + 1  \left( \binom{n+1}{2} - \binom{0}{2} \right) + 2  \left( \binom{n+1}{3} + \binom{1}{3} \right) \\
        \KnuthRFoldSum{1}{n}{3} &= 0 \left( \binom{n+1}{1} + \binom{-1}{1} \right) + 1 \left( \binom{n+1}{2} - \binom{0}{2} \right) + 6 \left( \binom{n+1}{3} + \binom{1}{3} \right) \\
        &+ 6 \left( \binom{n+1}{4} - \binom{2}{4} \right) \\
        \KnuthRFoldSum{1}{n}{m} &= \sum_{j=0}^{m} T(0, n, k) \left[ \binom{n+1}{j+1} + (-1)^j \binom{j-1}{j+1} \right]
    \end{align*}
