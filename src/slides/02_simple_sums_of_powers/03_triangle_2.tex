    \begin{align*}
        \KnuthRFoldSum{1}{n}{0} &= 1 \left( \binom{n-1}{1} + 1 \right) \\
        \KnuthRFoldSum{1}{n}{1} &= 2 \left( \binom{n-1}{1} + 1 \right) + 1  \left( \binom{n-1}{2} - 1 \right) \\
        \KnuthRFoldSum{1}{n}{2} &= 4 \left( \binom{n-1}{1} + 1 \right) + 5  \left( \binom{n-1}{2} - 1 \right) + 2  \left( \binom{n-1}{3} + 1 \right) \\
        \KnuthRFoldSum{1}{n}{3} &= 8 \left( \binom{n-1}{1} + 1 \right) + 19 \left( \binom{n-1}{2} - 1 \right) + 18 \left( \binom{n-1}{3} + 1 \right) \\
        &+ 6 \left( \binom{n-1}{4} - 1 \right) \\
        \KnuthRFoldSum{1}{n}{m} &= \sum_{j=0}^{m} T(2, m,j) \left[  \binom{n-1}{j+1} + (-1)^j \binom{j+1}{j+1} \right]
    \end{align*}
